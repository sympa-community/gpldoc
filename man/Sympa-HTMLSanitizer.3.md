# NAME

Sympa::HTMLSanitizer - Sanitize HTML contents

# SYNOPSIS

    $hss = Sympa::HTMLSanitizer->new;

    $sanitized = $hss->sanitize_html($html);
    $sanitized = $hss->sanitize_html_file($file);
    $hss->sanitize_var($variable);
